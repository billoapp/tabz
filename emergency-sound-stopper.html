Use this tool to immediately stop all bell notification sounds in your Tabeza staff app.

Instructions:

  1. Keep this page open
  2. Go to your staff app tab
  3. Press F12 to open Developer Tools
  4. Click the "Console" tab
  5. Come back here and click "Copy Stop Command"

‚úÖ Command copied to clipboard! Now paste it in your staff app's console.

Manual Console Command:

Copy and paste this into your staff app's browser console:

// Emergency sound stopper
try {
  // Stop continuous bell sound
  if (window.continuousBellActive) {
    window.continuousBellActive = false;
    console.log('üîá Stopped continuous bell');
  }
  
  // Clear bell timeout
  if (window.continuousBellTimeout) {
    clearTimeout(window.continuousBellTimeout);
    window.continuousBellTimeout = null;
    console.log('üîá Cleared bell timeout');
  }
  
  // Stop custom audio
  if (window.continuousAlertAudio) {
    window.continuousAlertAudio.pause();
    window.continuousAlertAudio.currentTime = 0;
    window.continuousAlertAudio = null;
    console.log('üîá Stopped custom audio');
  }
  
  // Stop all audio elements on the page
  document.querySelectorAll('audio').forEach(audio => {
    audio.pause();
    audio.currentTime = 0;
  });
  
  console.log('üîá ALL SOUNDS STOPPED SUCCESSFULLY!');
  alert('‚úÖ All sounds have been stopped!');
} catch (error) {
  console.error('Error stopping sounds:', error);
  alert('‚ùå Error stopping sounds. Try refreshing the page.');
}
